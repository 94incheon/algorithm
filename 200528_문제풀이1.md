- 코드

```python
def route(cur, s, cnt): # 좌표(x, y), 현재까지의 이동거리, 횟수
    global N, sumV

    # 백트래킹 (최소거리sumV 보다 s 현재거리가 더 클때 바로 return)
    if s >= sumV:
        return

    # 고객 전부 만날때,
    if cnt == N:
        s += abs(cur[0] - end[0]) + abs(cur[1] - end[1])
        if sumV > s:
            sumV = s
        return

    for i in range(len(L)):
        if visited[i]:
            continue
        visited[i] = 1
        route(L[i], s + abs(cur[0] - L[i][0]) + abs(cur[1] - L[i][1]), cnt + 1)
        visited[i] = 0


T = int(input())
for tc in range(1, T+1):
    N = int(input()) # 5
    l = list(map(int, input().split())) 

    L = [] # 고객 좌표 들어갈 배열
    sumV = float('inf') # 최소값 구하기 위해 최대값으로 초기화


    # 회사, 집, 고객좌표 값 받아오기
    for i in range(N+2):
        if i == 0:
            start = (l[i], l[i + 1]) # 회사
        elif i == 1:
            end = (l[2*i], l[2*i + 1]) # 집
        else:
            L.append((l[2*i],l[2*i + 1])) # 고객 좌표

    visited = [0] * len(L)


    route(start, 0, 0)
    print(f'#{tc} {sumV}')
```
